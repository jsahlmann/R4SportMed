Die Funktion survfit() aus dem R-Paket survival nutzt die folgende Definition für ein Event: 0 = Zensur, 1 = Event.

Da muss man aufpassen.

# proc lifetest;
#   time t*c(1); * 1 = Zensur, 0 = Event;
# run;

df$CNSR_neu <- 1 - df$CNSR
sf <- survfit(Surv(AVAL, CNSR_neu) ~ TRTP, data = df)

Differenzierterer Plot

ggsurvplot(
  sf,
  data = df,
  size = 1,                 # change line size
  palette =
    c("green", "#E7B800", "#2E9FDF"),# custom color palettes
  pval = TRUE,              # Add p-value
  risk.table = TRUE,        # Add risk table
  risk.table.col = "strata",# Risk table color by groups
  legend.labs =
    c("Placebo", "High dose", "Low dose"),    # Change legend labels
  risk.table.height = 0.35, # Useful to change when you have multiple groups
  ggtheme = theme_bw()      # Change ggplot2 theme
)

Differenzierterer Plot mit Konfidenzbändern

ggsurvplot(
  sf,
  data = df,
  size = 1,                 # change line size
  conf.int = TRUE,
  palette =
    c("green", "#E7B800", "#2E9FDF"),# custom color palettes
  pval = TRUE,              # Add p-value
  risk.table = TRUE,        # Add risk table
  risk.table.col = "strata",# Risk table color by groups
  legend.labs =
    c("Placebo", "High dose", "Low dose"),    # Change legend labels
  risk.table.height = 0.35, # Useful to change when you have multiple groups
  ggtheme = theme_bw()      # Change ggplot2 theme
)
